$versions = $versionsRaw | ConvertFrom-Csv |
    Sort-Object { [datetime]::Parse($_.Date) }, { [System.Version]$_.Version } -Descending

$changelog = ""

for ($i = 0; $i -lt $versions.Count; $i++) {
    $current     = $versions[$i]
    $releaseDate = [datetime]::Parse($current.Date)

    $sinceDate = if ($i + 1 -lt $versions.Count) {
        [datetime]::Parse($versions[$i + 1].Date)
    } else {
        [datetime]"1970-01-01"
    }

    $commits = git log `
        --pretty=format:"%s" `
        --date=short `
        "--since=$($sinceDate.ToString('yyyy-MM-dd'))" `
        "--until=$($releaseDate.ToString('yyyy-MM-dd'))"

    $changelog += "## [$($current.Version)] - $($releaseDate.ToString('yyyy-MM-dd'))`n"

    if ($commits) {
        $commits -split "`n" | ForEach-Object {
            $changelog += "- $_`n"
        }
    } else {
        $changelog += "- (no commits logged)`n"
    }

    $changelog += "`n"
}

$changelog
